```dax
// 💰 Year-To-Date Sales
YTD Total Sales = TOTALYTD(SUM(car_data[Price ($)]),'Calendar Table'[Date])

// 📅 Previous Year-To-Date Sales  
PYTD Total Sales = CALCULATE(SUM(car_data[Price ($)]),
                   DATESYTD(DATEADD('Calendar Table'[Date],-1,YEAR)))

// 📈 Sales Difference YTD vs PYTD
Sales Difference = [YTD Total Sales] - [PYTD Total Sales]

// 🚗 YTD Cars Sold
YTD Cars Sold = TOTALYTD(COUNT(car_data[Car_id]),'Calendar Table'[Date])

// 💵 YTD Average Price
YTD Avg Price = TOTALYTD(AVERAGE(car_data[Price ($)]),'Calendar Table'[Date])
